\begin{algorithm}
\caption{\textbf{CO\_Reception}( Message m\{Emetteur e, ListPred pred\})}
/* \textit{ où pred est la liste des prédécesseurs */}
\label{algo:recherche}
\begin{algorithmic}[1]
  \IF{ $\forall$ p$\in$ pred $\mid$ p.sn $\leq$ del(p.i)}
     \STATE  del(e.i) ++
     \STATE  cb +=\{e.sn, e.i\}
   \STATE  CO\_Deliver(m)
   \FORALL{ m\'~ $\in$ ListMessages}
   CO\_Receive(m\'~)
   \ENDFOR
   \ELSE
   \STATE ListMessages.add(m)
  \ENDIF
 \end{algorithmic}
\end{algorithm}

\caption{\textbf{CO\_Receive}( Message m\{Emetteur,ListPred\})}

   \ENDFOR
   \ELSE

  \ENDIF
